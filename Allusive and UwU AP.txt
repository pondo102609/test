-- ─── Utilities ───────────────────────────────────────────────────────────────

-- Bug fix: original could return empty string when second random < first index.
-- Using byte-based approach to guarantee a single character per iteration.
local function getRandomName()
    local chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    local len   = #chars
    local name  = {}
    for i = 1, 20 do
        name[i] = chars:sub(math.random(1, len), math.random(1, len))
    end
    return table.concat(name)
end

local function tween(instance, info, props)
    return Services.TweenService:Create(instance, info, props)
end

local QUINT_OUT = function(t) return TweenInfo.new(t or 0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out) end
local QUINT_IN  = function(t) return TweenInfo.new(t or 0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.In) end

local function convertStringToTable(str)
    local result = {}
    for value in str:gmatch("([^,]+)") do
        table.insert(result, value:match("^%s*(.-)%s*$"))
    end
    return result
end

local function convertTableToString(t)
    return table.concat(t, ", ")
end

-- ─── Connection Manager ───────────────────────────────────────────────────────

local function createConnections()
    local self = {}

    function self:disconnect(key)
        if not self[key] then return end
        self[key]:Disconnect()
        self[key] = nil
    end

    function self:disconnectAll()
        for key, value in pairs(self) do
            if typeof(value) == "RBXScriptConnection" then
                pcall(function() value:Disconnect() end)
                self[key] = nil
            end
        end
    end

    return self
end

-- ─── Config Manager ──────────────────────────────────────────────────────────

local function createConfig(basePath)
    local path = basePath or "Pulse"

    -- If executor file APIs aren't present, return a no-op stub
    if not isfolder then
        return {
            save = function() end,
            load = function()
                return { _flags = {}, _keybinds = {}, _library = {} }
            end
        }
    end

    if not isfolder(path) then
        pcall(function() makefolder(path) end)
    end

    local Config = {}

    function Config:save(fileName, data)
        if not writefile then return end
        local ok, err = pcall(function()
            writefile(path .. "/" .. fileName .. ".json", Services.HttpService:JSONEncode(data))
        end)
        if not ok then warn("[PulseUI] Config save failed:", err) end
    end

    function Config:load(fileName)
        local blank = { _flags = {}, _keybinds = {}, _library = {} }
        if not isfile or not readfile then return blank end

        local filePath = path .. "/" .. fileName .. ".json"
        local ok, result = pcall(function()
            if not isfile(filePath) then
                self:save(fileName, blank)
                return blank
            end
            local raw = readfile(filePath)
            if not raw or raw == "" then
                self:save(fileName, blank)
                return blank
            end
            return Services.HttpService:JSONDecode(raw)
        end)

        if not ok then
            warn("[PulseUI] Config load failed:", result)
            return blank
        end

        return result or blank
    end

    return Config
end

        local uis = Services.UserInputService
        if not uis.TouchEnabled and uis.KeyboardEnabled and uis.MouseEnabled then
            self._device = "PC"
        elseif uis.TouchEnabled then
            self._device = "Mobile"
        elseif uis.GamepadEnabled then
            self._device = "Console"
        else
            self._device = "Unknown"
        end
    end

    function Library:flagType(flag, typeName)
        return self._config._flags[flag] ~= nil
            and typeof(self._config._flags[flag]) == typeName
    end

    -- ── Dragging ─────────────────────────────────────────────────────────────

    local function onDragBegan(input)
        if input.UserInputType ~= Enum.UserInputType.MouseButton1
        and input.UserInputType ~= Enum.UserInputType.Touch then return end

        Library._dragging      = true
        Library._drag_start    = input.Position
        Library._container_pos = Container.Position

        local conn
        conn = input.Changed:Connect(function()
            if input.UserInputState ~= Enum.UserInputState.End then return end
            conn:Disconnect()
            Library._dragging = false
        end)
    end

    local function onDragMoved(input)
        if not Library._dragging then return end
        if input.UserInputType ~= Enum.UserInputType.MouseMovement
        and input.UserInputType ~= Enum.UserInputType.Touch then return end

        local delta = input.Position - Library._drag_start
        local pos   = Library._container_pos
        tween(Container, TweenInfo.new(0.1), {
            Position = UDim2.new(
                pos.X.Scale, pos.X.Offset + delta.X,
                pos.Y.Scale, pos.Y.Offset + delta.Y
            )
        }):Play()
    end

    Connections["drag_began"] = Container.InputBegan:Connect(onDragBegan)
    Connections["drag_moved"] = Services.UserInputService.InputChanged:Connect(onDragMoved)

    -- ── Tab Highlighting ─────────────────────────────────────────────────────
    -- Fix: was using object.TextLabel (name-based lookup) which breaks since
    -- all names are randomized. Now finds children by class/attribute instead.

    function Library:updateTabs(activeTab)
        for _, tab in ipairs(Tabs:GetChildren()) do
            if not tab:GetAttribute("IsTab") then continue end

            -- Find label and icon by class / attribute
            local label, icon
            for _, child in ipairs(tab:GetChildren()) do
                if child:IsA("TextLabel") then
                    label = child
                elseif child:IsA("ImageLabel") and child:GetAttribute("IsTabIcon") then
                    icon = child
                end
            end

            if tab == activeTab then
                -- Animate pin indicator to this tab's position
                local offset = (tab.LayoutOrder * 42) -- 38px tab + 4px padding
                local pinY   = 0.136 + (offset / 479)
                tween(PinIndicator, QUINT_OUT(0.45), {
                    Position = UDim2.new(0.026, 0, pinY, 0)
                }):Play()
                tween(tab,   QUINT_OUT(0.45), { BackgroundTransparency = 0.5 }):Play()
                if label then
                    tween(label, QUINT_OUT(0.45), {
                        TextTransparency = 0.2,
                        TextColor3 = Color3.fromRGB(152, 181, 255)
                    }):Play()
                end
                if icon then
                    tween(icon, QUINT_OUT(0.45), {
                        ImageTransparency = 0.2,
                        ImageColor3 = Color3.fromRGB(152, 181, 255)
                    }):Play()
                end
            else
                tween(tab, QUINT_OUT(0.45), { BackgroundTransparency = 1 }):Play()
                if label then
                    tween(label, QUINT_OUT(0.45), {
                        TextTransparency = 0.7,
                        TextColor3 = Color3.fromRGB(255, 255, 255)
                    }):Play()
                end
                if icon then
                    tween(icon, QUINT_OUT(0.45), {
                        ImageTransparency = 0.8,
                        ImageColor3 = Color3.fromRGB(255, 255, 255)
                    }):Play()
                end
            end
        end
    end

    function Library:updateSections(left, right)
        for _, section in ipairs(SectionsFolder:GetChildren()) do
            section.Visible = (section == left or section == right)
        end
    end

    -- ── Load ─────────────────────────────────────────────────────────────────

    function Library:Load()
        -- Preload images
        local images = {}
        for _, obj in ipairs(MenuGui:GetDescendants()) do
            if obj:IsA("ImageLabel") or obj:IsA("ImageButton") then
                table.insert(images, obj)
            end
        end
        Services.ContentProvider:PreloadAsync(images)

        Library:getDevice()

        if Library._device ~= "PC" then
            Library:getScreenScale()
            UIScale.Scale = Library._ui_scale
            Connections["viewport"] = Services.Workspace.CurrentCamera
                :GetPropertyChangedSignal("ViewportSize"):Connect(function()
                    Library:getScreenScale()
                    UIScale.Scale = Library._ui_scale
                end)
        end

        tween(Container, QUINT_OUT(0.5), { Size = UDim2.fromOffset(698, 479) }):Play()
        Library._ui_loaded = true
    end

        -- ── Module toggle logic ───────────────────────────────────────────────

        local function setModuleState(state)
            ModuleState.active = state

            local targetH = state and (93 + ModuleState.size) or 93
            tween(Module, QUINT_OUT(0.5), { Size = UDim2.fromOffset(241, targetH) }):Play()

            local toggleColor  = state and Color3.fromRGB(152, 181, 255) or Color3.fromRGB(0, 0, 0)
            local circleColor  = state and Color3.fromRGB(152, 181, 255) or Color3.fromRGB(66, 80, 115)
            local circleXScale = state and 0.53 or 0

            tween(Toggle, QUINT_OUT(0.5), { BackgroundColor3 = toggleColor }):Play()
            tween(Circle, QUINT_OUT(0.5), {
                BackgroundColor3 = circleColor,
                Position         = UDim2.new(circleXScale, 0, 0.5, 0)
            }):Play()

            Library._config._flags[settings.flag] = state
            Library:saveConfig()

            if settings.callback then
                settings.callback(state)
            end
        end

        -- Restore saved state
        if Library:flagType(settings.flag, "boolean") then
            ModuleState.active = Library._config._flags[settings.flag]
            if ModuleState.active then
                Toggle.BackgroundColor3 = Color3.fromRGB(152, 181, 255)
                Circle.BackgroundColor3 = Color3.fromRGB(152, 181, 255)
                Circle.Position         = UDim2.new(0.53, 0, 0.5, 0)
            end
            if settings.callback then settings.callback(ModuleState.active) end
        end

        Header.MouseButton1Click:Connect(function()
            setModuleState(not ModuleState.active)
        end)

        -- Helper: grow the options area by `amount` pixels
        local function growOptions(amount)
            if ModuleState.size == 0 then
                ModuleState.size = 11 -- initial padding

            if ModuleState.active then
                Module.Size = UDim2.fromOffset(241, 93 + ModuleState.size)
            end
            Options.Size = UDim2.fromOffset(241, ModuleState.size)
        end
