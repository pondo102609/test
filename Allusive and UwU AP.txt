-- Utility functions
local function convertStringToTable(inputString)
	local result = {}
	for value in string.gmatch(inputString, "([^,]+)") do
		local trimmedValue = value:match("^%s*(.-)%s*$")
		table.insert(result, trimmedValue)
	end
	return result
end

local function convertTableToString(inputTable)
	return table.concat(inputTable, ", ")
end

-- Connections handler
local function createConnections()
	return setmetatable({
		disconnect = function(self, connection)
			if not self[connection] then return end
			self[connection]:Disconnect()
			self[connection] = nil
		end,
		disconnect_all = function(self)
			for _, value in pairs(self) do
				if typeof(value) == 'function' then continue end
				pcall(function() value:Disconnect() end)
			end
		end
	}, {})
end

-- Config handler
local function createConfig(basePath)
	local path = basePath or "Pulse"
	
	if not isfolder then
		return {
			_flags = {},
			_keybinds = {},
			_library = {}
		}
	end
	
	if not isfolder(path) then
		pcall(function() makefolder(path) end)
	end
	
	return setmetatable({
		save = function(self, file_name, config)
			if not writefile then return end
			local success_save, result = pcall(function()
				local flags = Services.HttpService:JSONEncode(config)
				writefile(path .. '/' .. file_name .. '.json', flags)
			end)
			if not success_save then
				warn('Failed to save config:', result)
			end
		end,
		load = function(self, file_name, config)
			if not isfile or not readfile then return end
			local success_load, result = pcall(function()
				if not isfile(path .. '/' .. file_name .. '.json') then
					self:save(file_name, config)
					return
				end
				local flags = readfile(path .. '/' .. file_name .. '.json')
				if not flags then
					self:save(file_name, config)
					return
				end
				return Services.HttpService:JSONDecode(flags)
			end)
			if not success_load then
				warn('Failed to load config:', result)
			end
			if not result then
				result = {
					_flags = {},
					_keybinds = {},
					_library = {}
				}
			end
			return result
		end
	}, {})
end

	function Library:get_screen_scale()
		local viewport_size_x = Services.Workspace.CurrentCamera.ViewportSize.X
		self._ui_scale = viewport_size_x / 1400
	end
	
	-- Device detection
	function Library:get_device()
		local device = 'Unknown'
		if not Services.UserInputService.TouchEnabled and Services.UserInputService.KeyboardEnabled and Services.UserInputService.MouseEnabled then
			device = 'PC'
		elseif Services.UserInputService.TouchEnabled then
			device = 'Mobile'
		elseif Services.UserInputService.GamepadEnabled then
			device = 'Console'
		end
		self._device = device
	end
	
	-- Remove table value helper
	function Library:remove_table_value(__table, table_value)
		for index, value in pairs(__table) do
			if value ~= table_value then continue end
			table.remove(__table, index)
		end
	end
	
	-- Flag type checking
	function Library:flag_type(flag, flag_type)
		if not Library._config._flags[flag] then return end
		return typeof(Library._config._flags[flag]) == flag_type
	end

	-- Dragging functionality
	local function on_drag(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then 
			Library._dragging = true
			Library._drag_start = input.Position
			Library._container_position = Container.Position

			local endConnection
			endConnection = input.Changed:Connect(function()
				if input.UserInputState ~= Enum.UserInputState.End then return end
				endConnection:Disconnect()
				Library._dragging = false
			end)
		end
	end

	local function update_drag(input)
		local delta = input.Position - Library._drag_start
		local position = UDim2.new(
			Library._container_position.X.Scale, 
			Library._container_position.X.Offset + delta.X, 
			Library._container_position.Y.Scale, 
			Library._container_position.Y.Offset + delta.Y
		)
		Services.TweenService:Create(Container, TweenInfo.new(0.2), {
			Position = position
		}):Play()
	end

	Connections['container_input_began'] = Container.InputBegan:Connect(on_drag)
	
	Services.UserInputService.InputChanged:Connect(function(input)
		if not Library._dragging then return end
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			update_drag(input)
		end
	end)

	function Library:change_visibility(state)
		if state then
			Services.TweenService:Create(Container, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
				Size = UDim2.fromOffset(698, 479)
			}):Play()

				Size = UDim2.fromOffset(104.5, 52)
			}):Play()

	-- Tab management
	function Library:update_tabs(tab)
		for _, object in pairs(Tabs:GetChildren()) do
			if object.Name ~= 'Tab' then continue end
			if object == tab then
				if object.BackgroundTransparency ~= 0.5 then
					local offset = object.LayoutOrder * (0.113 / 1.3)
					Services.TweenService:Create(PinIndicator, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
						Position = UDim2.fromScale(0.026, 0.135 + offset)
					}):Play()    
					Services.TweenService:Create(object, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
						BackgroundTransparency = 0.5
					}):Play()
					Services.TweenService:Create(object.TextLabel, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
						TextTransparency = 0.2,
						TextColor3 = Color3.fromRGB(152, 181, 255)
					}):Play()
					Services.TweenService:Create(object.Icon, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
						ImageTransparency = 0.2,
						ImageColor3 = Color3.fromRGB(152, 181, 255)
					}):Play()
				end
				continue
			end
			if object.BackgroundTransparency ~= 1 then
				Services.TweenService:Create(object, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
					BackgroundTransparency = 1
				}):Play()
				Services.TweenService:Create(object.TextLabel, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
					TextTransparency = 0.7,
					TextColor3 = Color3.fromRGB(255, 255, 255)
				}):Play()
				Services.TweenService:Create(object.Icon, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
					ImageTransparency = 0.8,
					ImageColor3 = Color3.fromRGB(255, 255, 255)
				}):Play()
			end
		end
	end

	-- Load the UI
	function Library:Load()
		local content = {}
		for _, object in pairs(MenuGui:GetDescendants()) do
			if not object:IsA('ImageLabel') then continue end
			table.insert(content, object)
		end
		Services.ContentProvider:PreloadAsync(content)
		
		Library:get_device()

		if Library._device == 'Mobile' or Library._device == 'Unknown' then
			Library:get_screen_scale()
			UIScale.Scale = Library._ui_scale
			Connections['ui_scale'] = Services.Workspace.CurrentCamera:GetPropertyChangedSignal('ViewportSize'):Connect(function()
				Library:get_screen_scale()
				UIScale.Scale = Library._ui_scale
			end)
		end
	
		Services.TweenService:Create(Container, TweenInfo.new(0.5, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
			Size = UDim2.fromOffset(698, 479)
		}):Play()
		
		Library._ui_loaded = true
	end

					NumberSequenceKeypoint.new(0, 1),
					NumberSequenceKeypoint.new(0.5, 0),
					NumberSequenceKeypoint.new(1, 1)
				})
				Gradient.Parent = Divider
			end,
			
			CreateTextbox = function(textboxSettings)
				LayoutOrderModule = LayoutOrderModule + 1
				if ModuleManager._size == 0 then ModuleManager._size = 11 end
				ModuleManager._size = ModuleManager._size + 32
				if ModuleManager._state then
					Module.Size = UDim2.fromOffset(241, 93 + ModuleManager._size)

				Options.Size = UDim2.fromOffset(241, ModuleManager._size)

				Label.TextSize = 10

				Library:change_visibility(Library._ui_open)
			end
		end)
	end

	end)
